要实现一个登录界面获取本地IP的功能，查资料发现在 Vue3 中，可以通过组件的setup函数中使用window.location来获取IP,但是获取本机Ip不能够访问直接通过js在前端层面完成，出于安全原因，浏览器不允许直接访问本机IP.如果是在用户的本地网络上运行（如:通过localhost或特定的内网地址访问），所以使用WebRTC技术来获取本地IP地址。

pc.createDataChannel(‘’);

pc.createOffer().then(sdp => pc.setLocalDescription(sdp)).catch(console.error);

pc.onicecandidate = ice => {

if (ice && ice.candidate && ice.candidate.candidate) {

localIP.value = s[4];

}

}

}
